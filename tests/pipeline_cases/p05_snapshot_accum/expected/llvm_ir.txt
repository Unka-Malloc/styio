; ModuleID = 'styio'
source_filename = "styio"
target datalayout = "e-m:o-i64:64-i128:128-n32:64-S128"

@styio_str = private unnamed_addr constant [31 x i8] c"/tmp/styio_pipeline_factor.txt\00", align 1
@styio_fe_lit = private constant [3 x i64] [i64 1, i64 2, i64 3]
@styio_str.1 = private unnamed_addr constant [31 x i8] c"/tmp/styio_pipeline_factor.txt\00", align 1
@styio_print_at = private unnamed_addr constant [2 x i8] c"@\00", align 1
@styio_fmt_i64 = private unnamed_addr constant [6 x i8] c"%lld\0A\00", align 1

define i32 @main() {
main_entry:
  %prod = alloca i64, align 8
  %sum = alloca i64, align 8
  %factor = alloca i64, align 8
  %0 = call i64 @styio_read_file_i64line(ptr @styio_str)
  store i64 %0, ptr %factor, align 8
  store i64 0, ptr %sum, align 8
  %fe_idx = alloca i64, align 8
  store i64 0, ptr %fe_idx, align 8
  br label %foreach_hdr

foreach_exit:                                     ; preds = %foreach_hdr
  %1 = load i64, ptr %sum, align 8
  %2 = icmp eq i64 %1, -9223372036854775808
  br i1 %2, label %print_at, label %print_i64

foreach_hdr:                                      ; preds = %foreach_step, %main_entry
  %3 = load i64, ptr %fe_idx, align 8
  %4 = icmp slt i64 %3, 3
  br i1 %4, label %foreach_body, label %foreach_exit

foreach_body:                                     ; preds = %foreach_hdr
  %5 = load i64, ptr %fe_idx, align 8
  %6 = getelementptr inbounds [3 x i64], ptr @styio_fe_lit, i32 0, i64 %5
  %7 = load i64, ptr %6, align 8
  %x = alloca i64, align 8
  store i64 %7, ptr %x, align 8
  %8 = call i64 @styio_read_file_i64line(ptr @styio_str.1)
  store i64 %8, ptr %factor, align 8
  %9 = load i64, ptr %x, align 8
  %10 = load i64, ptr %factor, align 8
  %11 = icmp eq i64 %9, -9223372036854775808
  %12 = icmp eq i64 %10, -9223372036854775808
  %13 = or i1 %11, %12
  %14 = mul i64 %9, %10
  %15 = select i1 %13, i64 -9223372036854775808, i64 %14
  store i64 %15, ptr %prod, align 8
  %16 = load i64, ptr %sum, align 8
  %17 = load i64, ptr %prod, align 8
  %18 = load i64, ptr %sum, align 8
  %19 = load i64, ptr %prod, align 8
  %20 = add i64 %18, %19
  store i64 %20, ptr %sum, align 8
  br label %foreach_step

foreach_step:                                     ; preds = %foreach_body
  %21 = load i64, ptr %fe_idx, align 8
  %22 = add i64 %21, 1
  store i64 %22, ptr %fe_idx, align 8
  br label %foreach_hdr

print_at:                                         ; preds = %foreach_exit
  %23 = call i32 @puts(ptr @styio_print_at)
  br label %print_done

print_i64:                                        ; preds = %foreach_exit
  %24 = call i32 (ptr, ...) @printf(ptr @styio_fmt_i64, i64 %1)
  br label %print_done

print_done:                                       ; preds = %print_i64, %print_at
  ret i32 0
}
